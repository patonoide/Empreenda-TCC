CREATE TABLE avalia(
IDAVALIA SERIAL NOT NULL PRIMARY KEY,
COMENTARIO VARCHAR,

EMAIL VARCHAR, 
NOME VARCHAR NOT NULL);


CREATE TABLE categoria (
IDCATEGORIA SERIAL NOT NULL PRIMARY KEY,
NOME VARCHAR NOT NULL,
DESCRICAO VARCHAR
);
CREATE TABLE franquiador (
CNPJ CHAR(14) NOT NULL PRIMARY KEY,
CONTATO VARCHAR NOT NULL,
NOME  VARCHAR NOT NULL,
SENHA VARCHAR NOT NULL

);

CREATE TABLE franquia (
IDFRANQUIA SERIAL NOT NULL PRIMARY KEY,
NOME VARCHAR NOT NULL,
VALORTOTAL FLOAT NOT NULL,
VALORINICIAL FLOAT NOT NULL,
TEMPOLUCRO INTEGER NOT NULL,
DESCRICAO VARCHAR NOT NULL,
CAPITAL INTEGER NOT NULL,
CNPJ CHAR(14) NOT NULL,

FOTO VARCHAR(255),

FOREIGN KEY(CNPJ) REFERENCES franquiador (CNPJ)
);

CREATE TABLE empreendedor (
CPF CHAR(11) NOT NULL PRIMARY KEY,
NOME VARCHAR NOT NULL,
IDADE INTEGER NOT NULL,
DINHEIRO FLOAT NOT NULL,
CONTATO VARCHAR NOT NULL,
SENHA VARCHAR NOT NULL
);



CREATE TABLE pertencer (
IDCATEGORIA SERIAL NOT NULL,
IDFRANQUIA SERIAL NOT NULL,
FOREIGN KEY(IDCATEGORIA) REFERENCES categoria (IDCATEGORIA),
FOREIGN KEY(IDFRANQUIA) REFERENCES franquia (IDFRANQUIA)
);

CREATE TABLE possuir (
CPF CHAR(11) NOT NULL,
IDCATEGORIA SERIAL NOT NULL,
FOREIGN KEY(CPF) REFERENCES empreendedor (CPF),
FOREIGN KEY(IDCATEGORIA) REFERENCES categoria (IDCATEGORIA)
);
